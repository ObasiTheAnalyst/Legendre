\((1-x^2)y''-2xy'+n(n+1)y=0 \)
\( where\ n=1,2,3,...\)
\(y=C_1\ P_n(x)+ C_2\ Q_n(x)\)
\( where: \)
\(P_n(x)=\frac {1} {2^n}\sum_{r=0}^\frac {n} {2} \frac {(-1)^r\ (2n-2r)!} {(n-r)!\ (n-2r)!\ r!}x^{n-2r}\)
\(Q_n(x)=[P_n(x)]\ tanh^{-1}(x) \ \ \ \ \ +\)
\(\frac {1} {2^n}\sum_{r=0}^\frac {n-1} {2} \ (-1)^{r+1}\)\(\sum_{k=0}^r \frac {(2n-2k)!\ (2k)!}{((n-k)!\ k!)^2}\)\(\frac {(n-1-r)!\ r!}{(2r+1)!\ (n-1-2r)!}x^{n-1-2r}\)
